// 1. Newton Second Law
// ALGORITMA
1. TENTUKAN m DENGAN 600
2. TENTUKAN a DENGAN 2
3. TENTUKAN hasil
4. HITUNG m DIKALI a DISIMPAN DI hasil
5. TAMPILKAN hasil

// PSEUDOCODE
SET m WITH 600
SET a WITH 2
SET hasil
CALCULATE m TIMES a STORE IN hasil
DISPLAY hasil

// 2. Tahun Kabisat
// ALGORITMA
1. TENTUKAN tahun
2. BANDINGKAN tahun JIKA HABIS DIBAGI 4 DAN tahun TIDAK HABIS DIBAGI 100 MAKA TAMPILKAN 'kabisat'
3. BANDINGKAN tahun JIKA HABIS DIBAGI 4 DAN tahun HABIS DIBAGI 100 DAN tahun HABIS DIBAGI 400 MAKA TAMPILKAN 'kabisat'
4. SELAIN ITU TAMPILKAN 'bukan kabisat'

// PSEUDOCODE
SET tahun
IF tahun MODULUS 4 EQUALS 0 AND tahun MODULUS 100 NOT EQUALS 0 THEN
    DISPLAY 'kabisat'
ELSE IF tahun MODULUS 4 EQUALS 0 AND tahun MODULUS 100 EQUALS 0 AND tahun MODULUS 400 EQUALS 0 THEN
    DISPLAY 'kabisat'
ELSE
    DISPLAY 'bukan kabisat'
END IF

// 3. Laundry Day
// ALGORITMA
1. TENTUKAN JUMLAH pakaian YANG AKAN DIMASUKKAN KE MESIN CUCI YAITU 20
2. pakaian DIMASUKKAN SATU PERSATU KE DALAM MESIN CUCI
3. JIKA pakaian SUDAH MASUK SEMUA MAKA CETAK 'mesin cuci menyala' dan TAMPILKAN JUMLAH pakaian 

// PSEUDOCODE
SET pakaian EQUALS 0
FOR pakaian TO 20 INCREMENT 1
    CALCULATE pakaian PLUS 1 STORE IN pakaian
    IF pakaian EQUALS 20 THEN
        DISPLAY pakaian
    END IF
END FOR

// 4. Periksa Kuku
// ALGORITMA
1. TENTUKAN siswa DENGAN NILAI 40
2. guru BERKELILING DARI siswa KE-1 SAMPAI KE-40 DENGAN PENAMBAHAN 1
3. JIKA guru MENEMUKAN siswa MEMILIKI 'kuku panjang' MAKA TAMPILKAN 'dihukum'
4. SELAIN ITU TAMPILKAN 'tidak dihukum'
5. JIKA guru BELUM MENGELILINGI 40 siswa MAKA KEMBALI KE STEP 3

// PSEUDOCODE
SET siswa WITH 1
SET kuku_siswa
FOR siswa TO 40 INCREMENT 1
    IF kuku_siswa EQUALS 'kuku panjang' THEN
        DISPLAY 'Hukum Siswa'
    ELSE
        DISPLAY 'Puji Siswa'
    END IF
END FOR
